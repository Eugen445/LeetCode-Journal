#### [26. Remove Duplicates from Sorted Array](https://leetcode-cn.com/problems/remove-duplicates-from-sorted-array/)

```
class Solution {
public:
    int removeDuplicates(vector<int>& nums) {

        int n = nums.size();
        if (n == 0) return 0;

        int slow = 1, fast = 1;
        while (fast < n) {
            if (nums[fast] != nums[fast - 1])
                nums[slow++] = nums[fast];
            ++fast;
        }
        return slow;
    }
};

```
