#### [70. Climbing Stairs](https://leetcode-cn.com/problems/climbing-stairs/)

C++：

```
1.动态规划
class Solution {
public:
    int climbStairs(int n) {

        assert(n >= 0);
        if (n <= 2) return n;

        vector<int> dp(n + 1);
        dp[0] = 0, dp[1] = 1, dp[2] = 2;
        for (int i = 3; i <= n; ++i) {
            dp[i] = dp[i - 1] + dp[i - 2];
        }
        
        return dp[n];
    }
};
2.动态规划
class Solution {
public:
    int climbStairs(int n) {

        assert(n >= 0);
        if (n <= 2) return n;

        int dp[2] = {1, 2};
        for (int i = 3; i <= n; ++i) {
            int tmp = dp[1];
            dp[1] = dp[0] + dp[1];
            dp[0] = tmp;
        }

        return dp[1];
    }
};
```

python3：

```
class Solution:
    def climbStairs(self, n: int) -> int:
        if (n < 3): return n

        a, b, c = 1, 2, 0
        for i in range(3, n + 1):
            c = a + b
            a, b = b, c
        return b
```
