class Solution {
public:
    void sortColors(vector<int>& nums) {

        int zero = 0;
        int one = 0;
        int two = 0;
        for (int i = 0; i < nums.size(); ++i) {
            if (nums[i] == 0) ++zero;
            else if (nums[i] == 1) ++one;
            else ++two;
        }

        for (int i = 0; i < nums.size(); ++i) {
            if (i < zero) nums[i] = 0;
            else if (i < zero + one) nums[i] = 1;
            else nums[i] = 2;
        }
    }
};

class Solution {
public:
	void sortColors(vector<int>& nums) {

		int count[3] = { 0 };
		for (int i = 0; i < nums.size(); ++i) {
			assert(nums[i] >= 0 && nums[i] <= 2);
			count[nums[i]]++;
		}

		int index = 0;
		for (int i = 0; i < count[0]; ++i)
			nums[index++] = 0;
		for (int i = index; i < count[0] + count[1]; ++i)
			nums[index++] = 1;
		for (int i = index; i < nums.size(); ++i)
			nums[index++] = 2;
	}
};

class Solution {
public:
    void sortColors(vector<int>& nums) {

        int zero = -1;
        int two = nums.size();

        for (int i = 0; i < two; ) {
            if (nums[i] == 1) ++i;
            else if (nums[i] == 2)
                swap(nums[i], nums[--two]);
            else {
                assert(nums[i] == 0);
                swap(nums[i++], nums[++zero]);
            }
        }
    }
};
