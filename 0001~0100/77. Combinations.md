#### [77. Combinations](https://leetcode-cn.com/problems/combinations/)

```C++
1.回溯法
class Solution {
private:
    vector<vector<int>> res;
    vector<int> vec;
public:
    vector<vector<int>> combine(int n, int k) {
        dfs(n, k, 1);
        return res;
    }
private:
    void dfs(int n, int k, int startIndex) {

        if (vec.size() == k) {
            res.push_back(vec);
            return ;
        }

        for (int i = startIndex; i <= n; ++i) {
            vec.push_back(i);
            dfs(n, k, i + 1);
            vec.pop_back();
        }
    }
};
2.回溯剪枝
class Solution {
private:
    vector<vector<int>> res;
    vector<int> vec;
public:
    vector<vector<int>> combine(int n, int k) {
        dfs(n, k, 1);
        return res;
    }
private:
    void dfs(int n, int k, int startIndex) {

        if (vec.size() == k) {
            res.push_back(vec);
            return ;
        }

        for (int i = startIndex; i <= n - (k - (int)vec.size()) + 1; ++i) {
            vec.push_back(i);
            dfs(n, k, i + 1);
            vec.pop_back();
        }
    }
};
```
