## 136. Single Number
```C++
class Solution {
public:
    int singleNumber(vector<int>& nums) {

        if (nums.size() == 0) 
        throw invalid_argument("the input has no solution");
        if (nums.size() == 1)
        return nums[0];

        int res = 0;
        for (int i = 0; i < nums.size(); ++i) 
            res ^= nums[i];
        return res;
    }
};
```
