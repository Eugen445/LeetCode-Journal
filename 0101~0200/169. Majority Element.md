[169. Majority Element](https://leetcode-cn.com/problems/majority-element/)

1.哈希法

```
哈希.1
class Solution {
public:
    int majorityElement(vector<int>& nums) {
        unordered_map<int,int> umap;
        for (int num : nums)
            ++umap[num];
        int n = nums.size() / 2;
        for (int num : nums) {
            if (umap[num] > n) {
                return num;
            }
        }
        return {};
    }
};
哈希.2
class Solution {
public:
    int majorityElement(vector<int>& nums) {
        unordered_map<int,int> umap;
        int maxNum = 0; 
        int count = 0; //记录出现最多的数
        for (int num : nums) {
            ++umap[num];
            if (umap[num] > count) {
                count = umap[num];
                maxNum = num;
            }
        }
        return maxNum;
    }
};
```

2.排序

```
时间复杂度:O(nlogn)
class Solution {
public:
    int majorityElement(vector<int>& nums) {
        sort(nums.begin(), nums.end());
        return nums[nums.size() / 2];
    }
};
```

3.摩尔投票法

```
class Solution {
public:
    int majorityElement(vector<int>& nums) {
        int mostNum = nums[0];
        int count = 1;
        for (int i = 1; i < nums.size(); ++i) {
            if (nums[i] != mostNum) {
                --count;
                if (count == 0) {
                    mostNum = nums[i];
                    count = 1;
                }
            }
            else ++count;
        } 
        return mostNum;
    }
};
```
