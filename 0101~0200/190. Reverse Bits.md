#### [190. Reverse Bits](https://leetcode-cn.com/problems/reverse-bits/)

```C++
class Solution {
public:
    uint32_t reverseBits(uint32_t n) {
        int res = 0;
        for (int i = 0; i < 32 && n > 0; ++i) {
            res |= (n & 1) << (31 - i);
            n >>= 1;
        }
        return res;
    }
};
```
