#### [209. Minimum Size Subarray Sum](https://leetcode-cn.com/problems/minimum-size-subarray-sum/)

class Solution {
public:
    int minSubArrayLen(int target, vector<int>& nums) {

        int minLen = 10e5;
        for (int i = 0; i < nums.size(); ++i) {
            
            int sum = 0;
            int count = 0;
            for (int j = i; j < nums.size(); ++j) {
                ++count;
                sum += nums[j];
                if (sum >= target) {
                    minLen = min(minLen, count);
                    break;
                }
            }
        }

        return minLen == 10e5 ? 0 : minLen;
    }
};

class Solution {
public:
    int minSubArrayLen(int target, vector<int>& nums) {

        int l = 0, r = 0;
        int minLen = 10e5;
        int sum = 0;
        while (r < nums.size()) {
            sum += nums[r];
            if (sum >= target) {
                minLen = min(minLen, r - l + 1);
                while (l < r) {
                    sum -= nums[l++];
                    if (sum >= target)
                        minLen = min(minLen, r - l + 1);
                    else break;
                }
            }
            ++r;
        }

        return minLen == 10e5 ? 0 : minLen;
    }
};

class Solution {
public:
    int minSubArrayLen(int target, vector<int>& nums) {

        int result = 10e5;
        int size = nums.size();
        int len = 0;
        int sum = 0;
        int i = 0;

        for (int j = 0; j < size; j++){
            sum += nums[j];

            while (sum >= target){
                len = (j - i + 1);
                result = result < len ? result : len;
                sum -= nums[i++];
            }
        }
        return result == 10e5 ? 0 : result;
    }
};

class Solution {
public:
    int minSubArrayLen(int target, vector<int>& nums) {

        int l = 0, r = -1;  // nums[l...r]为滑动窗口
        int sum = 0;
        int res = nums.size() + 1;

        while (l < nums.size()) {

            if (r + 1 < nums.size() && sum < target)
                sum += nums[++r];
            else 
                sum -= nums[l++];

            if (sum >= target)
                res = min(res, r - l + 1);
        }

        return res == nums.size() + 1 ? 0 : res;
    }
};
