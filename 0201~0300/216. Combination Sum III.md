#### [216. Combination Sum III](https://leetcode-cn.com/problems/combination-sum-iii/)

```C++
1.回溯法
class Solution {
private:
    vector<vector<int>> res;
    vector<int> path;
public:
    vector<vector<int>> combinationSum3(int k, int n) {
        recall(k, n, 1, 0);
        return res;
    }
private:
    void recall(int k, int n, int index, int sum) {
        if (path.size() == k) {
            if (sum == n) {
                res.push_back(path);
            }
            return ;
        }

        //for (int i = 1; i <= 9; ++i) {
        for (int i = index; i <= 9; ++i) {
            path.push_back(i);
            sum += i;
            recall(k, n, i + 1, sum;
            sum -= i;
            path.pop_back();
        }
    }
};
2.回溯剪枝
class Solution {
private:
    vector<vector<int>> res;
    vector<int> path;
public:
    vector<vector<int>> combinationSum3(int k, int n) {
        recall(k, n, 1, 0);
        return res;
    }
private:
    void recall(int k, int n, int index, int sum) {
        if (sum > n) return ;

        if (path.size() == k) {
            if (sum == n) {
                res.push_back(path);
            }
            return ;
        }

        //for (int i = 1; i <= 9; ++i) {
        //for (int i = index; i <= 9 && i <= n; ++i) { //ok但不是最好
        for(int i = index; i <= 9 - (k - path.size()) + 1; ++i) {
            path.push_back(i);
            sum += i;
            recall(k, n, i + 1, sum);
            sum -= i;
            path.pop_back();
        }
    }
};
```
