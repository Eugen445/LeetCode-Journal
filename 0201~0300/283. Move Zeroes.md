#### [283. Move Zeroes](https://leetcode-cn.com/problems/move-zeroes/)

class Solution {
public:
    void moveZeroes(vector<int>& nums) {

        int n = nums.size();
        int left = 0, right = n - 1;
        while (left <= right) {
            if (nums[left] == 0) {
                for (int j = left; j < right; ++j)
                    nums[j] = nums[j + 1];
                nums[right--] = 0;
            }
            else ++left;
        }
    }
};

class Solution {
public:
    void moveZeroes(vector<int>& nums) {

        int l = 0, r = nums.size();
        for (int i = 0; i < r; ++i) 
            if (nums[i] != 0) 
                nums[l++] = nums[i];
        for (int i = l; i < r; ++i)
            nums[i] = 0;
    }
};

class Solution {
public:
    void moveZeroes(vector<int>& nums) {

        int n = nums.size(), l = 0, r = 0;
        while (r < n) {
            if (nums[r]) { //全为非0元素时,交换是没有意义的
                swap(nums[l], nums[r]);
                ++l;
            }
            ++r;
        }
    }
};
