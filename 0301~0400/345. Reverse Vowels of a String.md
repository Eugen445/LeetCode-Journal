#### [345. Reverse Vowels of a String](https://leetcode-cn.com/problems/reverse-vowels-of-a-string/)

```
class Solution {
public:
    string reverseVowels(string s) {

        int l = 0, r = s.size() - 1;
        while ( l < r) {
            
            while (!judge(s[l]) && l < r) ++l;
            while (!judge(s[r]) && l < r) --r;

            if (l == r) break;
            swap(s[l++], s[r--]);
        }

        return s;
    }
private:
    bool judge(char e) {

        char aeiou[] = {'a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U'};
        for (int i = 0; i < 10; ++i)
            if (e == aeiou[i])
                return true;
        return false;
    }
};
```
