#### [412. Fizz Buzz](https://leetcode-cn.com/problems/fizz-buzz/)

1.
class Solution {
public:
    vector<string> fizzBuzz(int n) {

        vector<string> res;
        string aux3_5 = "FizzBuzz";
        string aux3 = "Fizz";
        string aux5 = "Buzz";
        for (int i = 1; i <= n; ++i) {
            if ((i % 3 == 0) && (i % 5 == 0))
                res.push_back(aux3_5);
            else if (i % 3 == 0)
                res.push_back(aux3);
            else if (i % 5 == 0)
                res.push_back(aux5);
            else   
                res.push_back(to_string(i));
        }

        return res;
    }
};
