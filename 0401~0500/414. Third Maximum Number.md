#### [414. Third Maximum Number](https://leetcode-cn.com/problems/third-maximum-number/)

class Solution {
public:
    int thirdMax(vector<int>& nums) {

        int min_val = INT_MAX;
        for (int e : nums)
            min_val = min(e, min_val);
        int max_one = min_val;
        int max_two = min_val;
        int max_three = min_val;

        for (int i = 0; i < nums.size(); ++i) {
            if (nums[i] > max_one) {
                max_three = max_two;
                max_two = max_one;
                max_one = nums[i];
            }
            else if (nums[i] > max_two && nums[i] != max_one) {
                max_three = max_two;
                max_two = nums[i];
            }
            else if (nums[i] > max_three && nums[i] != max_two && nums[i] != max_one) {
                max_three = nums[i];
            }
            //cout << "max_one = " << max_one << endl;
            //cout << "max_two = " << max_two << endl;
            //cout << "max_three = " << max_three << endl;
        }

        if (max_one != max_two && max_two != max_three) return max_three;
        return max_one;
    }
};
