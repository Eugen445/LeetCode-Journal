#### [509. Fibonacci Number](https://leetcode-cn.com/problems/fibonacci-number/)

C++：

```C++
1.动态规划
class Solution {
public:
    int fib(int n) {

        if (n <= 1) return n;
        vector<int> dp(n + 1); //1.下标的含义
        dp[0] = 0, dp[1] = 1; //3.初始化

        for (int i = 2; i <= n; ++i) { //4.遍历顺序
            dp[i] = dp[i - 1] + dp[i - 2]; //2.推导公式
            //cout << "i = " << dp[i] << "\n"; //5.打印数组验证
        }

        return dp[n];
    }
};

2.动态规划
class Solution {
public:
    int fib(int n) {

        if (n < 2) return n;

        int dp[2] = {0};
        dp[0] = 0, dp[1] = 1;
        for (int i = 2; i <= n; ++i) {
            int tmp = dp[1];
            dp[1] = dp[0] + dp[1];
            dp[0] = tmp;
        }

        return dp[1];
    }
};
3.递归版
class Solution {
public:
    int fib(int n) {

        if (n < 2) return n;
        return fib(n - 1) + fib(n - 2);
    }
};
```

python3：

```python
class Solution:
    def fib(self, n: int) -> int:
        if n < 2:
            return n
        a, b, c = 0, 1, 0
        for i in range(2, n + 1):
            c = a + b
            a, b = b, c
        return c
```
