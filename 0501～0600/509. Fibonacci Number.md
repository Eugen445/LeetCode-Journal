#### [509. Fibonacci Number](https://leetcode-cn.com/problems/fibonacci-number/)

```
1.动态规划
class Solution {
public:
    int fib(int n) {

        if (n < 2) return n;
        int dp[2] = {0, 1};

        for (int i = 2; i <= n; ++i) {

            int tmp = dp[1];
            dp[1] = dp[1] + dp[0];
            dp[0] = tmp;
        }

        return dp[1];
    }
};
```
