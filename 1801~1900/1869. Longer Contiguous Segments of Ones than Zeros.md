1.两次遍历
class Solution {
public:
    bool checkZeroOnes(string s) {

        int ones = 0;
        int count1 = 0;
        for (int i = 0; i < s.size(); ++i) {
            if (s[i] == '1') {
                ++count1;
                ones = max(ones, count1);
            } else count1 = 0;
        }
        int zeros = 0;
        int count0 = 0;
        for (int i = 0; i < s.size(); ++i) {
            if (s[i] == '0') {
                ++count0;
                zeros = max(zeros, count0);
            } else count0 = 0;
        }
        
        return ones > zeros;
    }
};
